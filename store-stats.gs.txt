// CÁCH CÀI ĐẶT LẦN ĐẦU (nếu bạn CHƯA từng cài file này):
// 1. Vào sheets.google.com, tạo 1 Google Sheet MỚI (trống), đặt tên gì cũng
//    được (ví dụ "MeoMoney Store Stats"). Không cần tạo sẵn tab/cột gì.
// 2. Trong sheet đó: Tiện ích mở rộng > Apps Script.
// 3. Xóa hết code mẫu (myFunction rỗng), dán TOÀN BỘ nội dung file này vào.
// 4. Vào Triển khai (Deploy) > Triển khai mới (New deployment):
//    - Loại: Ứng dụng web (Web app)
//    - Execute as: Me (email của bạn)
//    - Who has access: Anyone (bắt buộc, để trang store.html gọi vào được)
//    Bấm Triển khai, copy URL dạng https://script.google.com/macros/s/xxxx/exec
// 5. Gửi URL đó lại cho Claude để gắn vào store.html (biến STORE_STATS_URL).
//
// NẾU BẠN ĐÃ CÀI FILE NÀY TRƯỚC ĐÓ RỒI (chỉ đang nâng cấp thêm Bình luận):
// - Vào lại Apps Script cũ, XÓA HẾT code đang có, dán TOÀN BỘ nội dung file
//   này (bản mới) đè lên.
// - Vào Triển khai (Deploy) > Quản lý bản triển khai (Manage deployments) >
//   bấm icon bút chì (✏️) ở bản đang chạy > mục "Phiên bản/Version" chọn
//   "New version" > Triển khai (Deploy). URL giữ nguyên, không cần gửi lại.
//
// Mở sheet lên bất cứ lúc nào để xem tab "Downloads" (mã máy đã tải, không
// trùng lặp), "Ratings" (mã máy + số sao) và "Comments" (bình luận của khách).

const DOWNLOADS_SHEET = 'Downloads';
const RATINGS_SHEET = 'Ratings';
const COMMENTS_SHEET = 'Comments';
const SO_BINH_LUAN_HIEN_THI = 30; // chỉ trả về N bình luận mới nhất cho trang store

function laySheetTai() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(DOWNLOADS_SHEET);
  if (!sheet) {
    sheet = ss.insertSheet(DOWNLOADS_SHEET);
    sheet.appendRow(['Thời gian', 'Mã máy']);
  }
  return sheet;
}

function laySheetDanhGia() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(RATINGS_SHEET);
  if (!sheet) {
    sheet = ss.insertSheet(RATINGS_SHEET);
    sheet.appendRow(['Thời gian', 'Mã máy', 'Sao']);
  }
  return sheet;
}

function laySheetBinhLuan() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(COMMENTS_SHEET);
  if (!sheet) {
    sheet = ss.insertSheet(COMMENTS_SHEET);
    sheet.appendRow(['Thời gian', 'Mã máy', 'Tên', 'Nội dung']);
  }
  return sheet;
}

// Ghi 1 lượt tải, chống đếm trùng: mỗi mã máy chỉ tính 1 lần duy nhất.
function ghiLuotTai(deviceId) {
  if (!deviceId) return;
  const sheet = laySheetTai();
  const data = sheet.getDataRange().getValues();
  const daCo = data.slice(1).some((r) => String(r[1]).trim().toUpperCase() === deviceId);
  if (!daCo) sheet.appendRow([new Date(), deviceId]);
}

// Ghi/sửa đánh giá sao của 1 mã máy. Nếu máy đó đã đánh giá trước đó thì
// cập nhật lại số sao mới (mỗi máy chỉ tính 1 đánh giá, đổi ý thì ghi đè).
function ghiDanhGia(deviceId, sao) {
  if (!deviceId || !sao || sao < 1 || sao > 5) return;
  const sheet = laySheetDanhGia();
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][1]).trim().toUpperCase() === deviceId) {
      sheet.getRange(i + 1, 1, 1, 3).setValues([[new Date(), deviceId, sao]]);
      return;
    }
  }
  sheet.appendRow([new Date(), deviceId, sao]);
}

// Ghi 1 bình luận mới. Tên để trống thì hiện "Người dùng ẩn danh" ở phía
// trang web. Giới hạn độ dài để tránh spam quá dài.
function ghiBinhLuan(deviceId, ten, noiDung) {
  noiDung = String(noiDung || '').trim().slice(0, 300);
  if (!deviceId || !noiDung) return;
  ten = String(ten || '').trim().slice(0, 40);
  const sheet = laySheetBinhLuan();
  sheet.appendRow([new Date(), deviceId, ten, noiDung]);
}

// Lấy N bình luận mới nhất (mới nhất lên đầu) để hiển thị trên trang store.
function layBinhLuan() {
  const data = laySheetBinhLuan().getDataRange().getValues().slice(1);
  return data
    .slice(-SO_BINH_LUAN_HIEN_THI)
    .reverse()
    .map((r) => ({ ten: String(r[2] || ''), noiDung: String(r[3] || '') }));
}

function tinhThongKe() {
  const dataTai = laySheetTai().getDataRange().getValues().slice(1);
  const dataDanhGia = laySheetDanhGia().getDataRange().getValues().slice(1);

  const downloads = dataTai.length;
  const ratingCount = dataDanhGia.length;
  const tongSao = dataDanhGia.reduce((tong, r) => tong + (Number(r[2]) || 0), 0);
  const avgRating = ratingCount > 0 ? Math.round((tongSao / ratingCount) * 10) / 10 : 0;

  return { downloads, avgRating, ratingCount, comments: layBinhLuan() };
}

// Web App: trang store.html gọi vào đây, ví dụ:
//   ?action=download&deviceId=ABC123                     -> ghi 1 lượt tải
//   ?action=rate&deviceId=ABC123&stars=5                  -> ghi đánh giá 5 sao
//   ?action=comment&deviceId=ABC123&name=Nam&text=Hay!    -> ghi 1 bình luận
//   (không truyền action)                                 -> chỉ lấy dữ liệu hiện tại
// Mọi trường hợp đều trả về { downloads, avgRating, ratingCount, comments }.
function doGet(e) {
  const params = (e && e.parameter) || {};
  const action = params.action || '';
  const deviceId = String(params.deviceId || '').trim().toUpperCase();

  if (action === 'download') {
    ghiLuotTai(deviceId);
  } else if (action === 'rate') {
    ghiDanhGia(deviceId, parseInt(params.stars, 10));
  } else if (action === 'comment') {
    ghiBinhLuan(deviceId, params.name, params.text);
  }

  const output = tinhThongKe();
  return ContentService.createTextOutput(JSON.stringify(output)).setMimeType(ContentService.MimeType.JSON);
}
