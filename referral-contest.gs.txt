var SHEET_ACTIVE = 'ActiveUsers';   // các mã đã là người dùng thật (whitelist)
var SHEET_PLAYERS = 'Players';      // tài khoản người chơi
var SHEET_REFERRALS = 'Referrals';  // mỗi mã đã có người nhận -> 1 dòng
var LEADERBOARD_SIZE = 30;

// ---------------- tiện ích chung ----------------

function getSheet_(name, headers) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(headers);
  }
  return sheet;
}

function json_(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function norm_(s) {
  return String(s == null ? '' : s).trim().toUpperCase();
}

function hashPass_(user, pass) {
  var raw = 'meomoney-contest::' + String(user).toLowerCase() + '::' + String(pass);
  var bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, raw, Utilities.Charset.UTF_8);
  var out = '';
  for (var i = 0; i < bytes.length; i++) {
    var b = (bytes[i] + 256) % 256;
    out += (b < 16 ? '0' : '') + b.toString(16);
  }
  return out;
}

function newToken_() {
  return Utilities.getUuid().replace(/-/g, '');
}

// ---------------- whitelist người dùng thật ----------------

function recordActivity_(code) {
  code = norm_(code);
  if (!/^[A-Z0-9]{4,12}$/.test(code)) return;
  var sheet = getSheet_(SHEET_ACTIVE, ['Mã', 'Lần đầu', 'Gần nhất']);
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (norm_(data[i][0]) === code) {
      sheet.getRange(i + 1, 3).setValue(new Date());
      return;
    }
  }
  sheet.appendRow([code, new Date(), new Date()]);
}

function activeCodeSet_() {
  var sheet = getSheet_(SHEET_ACTIVE, ['Mã', 'Lần đầu', 'Gần nhất']);
  var data = sheet.getDataRange().getValues();
  var set = {};
  for (var i = 1; i < data.length; i++) set[norm_(data[i][0])] = true;
  return set;
}

// ---------------- điểm & bảng xếp hạng ----------------

// Trả về { username: soMaHopLe }. Chỉ đếm mã đang nằm trong whitelist.
function computeCounts_() {
  var active = activeCodeSet_();
  var ref = getSheet_(SHEET_REFERRALS, ['Mã', 'Người nhận', 'Thời gian']).getDataRange().getValues();
  var counts = {};
  for (var i = 1; i < ref.length; i++) {
    var code = norm_(ref[i][0]);
    var by = String(ref[i][1]);
    if (active[code]) counts[by] = (counts[by] || 0) + 1;
  }
  return counts;
}

function rankedList_() {
  var counts = computeCounts_();
  var arr = [];
  for (var u in counts) arr.push({ username: u, count: counts[u] });
  arr.sort(function (a, b) {
    if (b.count !== a.count) return b.count - a.count;
    return a.username.toLowerCase() < b.username.toLowerCase() ? -1 : 1;
  });
  return arr;
}

// ---------------- API (gọi từ trang web qua google.script.run) ----------------

function apiRegister(user, pass) {
  user = String(user == null ? '' : user).trim();
  pass = String(pass == null ? '' : pass);
  if (user.length < 2 || user.length > 24) return { ok: false, error: 'Tên hiển thị cần 2–24 ký tự.' };
  if (pass.length < 4) return { ok: false, error: 'Mật khẩu tối thiểu 4 ký tự.' };

  var lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    var sheet = getSheet_(SHEET_PLAYERS, ['Tên', 'MậtKhẩuHash', 'Token', 'Tạo lúc']);
    var data = sheet.getDataRange().getValues();
    var key = user.toLowerCase();
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).toLowerCase() === key) return { ok: false, error: 'Tên này đã có người dùng, chọn tên khác nhé.' };
    }
    var tok = newToken_();
    sheet.appendRow([user, hashPass_(user, pass), tok, new Date()]);
    return { ok: true, token: tok, username: user };
  } finally {
    lock.releaseLock();
  }
}

function apiLogin(user, pass) {
  user = String(user == null ? '' : user).trim();
  pass = String(pass == null ? '' : pass);
  var sheet = getSheet_(SHEET_PLAYERS, ['Tên', 'MậtKhẩuHash', 'Token', 'Tạo lúc']);
  var data = sheet.getDataRange().getValues();
  var key = user.toLowerCase();
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][0]).toLowerCase() === key) {
      if (String(data[i][1]) === hashPass_(String(data[i][0]), pass)) {
        return { ok: true, token: String(data[i][2]), username: String(data[i][0]) };
      }
      return { ok: false, error: 'Sai mật khẩu.' };
    }
  }
  return { ok: false, error: 'Tài khoản không tồn tại.' };
}

function playerByToken_(token) {
  token = String(token == null ? '' : token);
  if (!token) return null;
  var sheet = getSheet_(SHEET_PLAYERS, ['Tên', 'MậtKhẩuHash', 'Token', 'Tạo lúc']);
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (String(data[i][2]) === token) return { row: i + 1, username: String(data[i][0]) };
  }
  return null;
}

function apiSubmit(token, code) {
  var player = playerByToken_(token);
  if (!player) return { ok: false, error: 'need-login' };
  code = norm_(code);
  if (!/^[A-Z0-9]{4,12}$/.test(code)) return { ok: false, error: 'Mã không hợp lệ (chỉ gồm chữ và số).' };

  var lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    var sheet = getSheet_(SHEET_REFERRALS, ['Mã', 'Người nhận', 'Thời gian']);
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (norm_(data[i][0]) === code) {
        if (String(data[i][1]).toLowerCase() === player.username.toLowerCase()) return { ok: false, error: 'Bạn đã nhập mã này rồi.' };
        return { ok: false, error: 'Mã này đã được người khác nhập trước.' };
      }
    }
    sheet.appendRow([code, player.username, new Date()]);
    var isActive = activeCodeSet_()[code] === true;
    return { ok: true, valid: isActive, status: isActive ? 'valid' : 'pending' };
  } finally {
    lock.releaseLock();
  }
}

function apiLeaderboard() {
  var arr = rankedList_();
  return { ok: true, top: arr.slice(0, LEADERBOARD_SIZE), totalPlayers: arr.length };
}

function apiMe(token) {
  var player = playerByToken_(token);
  if (!player) return { ok: false, error: 'need-login' };
  var active = activeCodeSet_();
  var ref = getSheet_(SHEET_REFERRALS, ['Mã', 'Người nhận', 'Thời gian']).getDataRange().getValues();
  var codes = [];
  var validCount = 0;
  for (var i = 1; i < ref.length; i++) {
    if (String(ref[i][1]).toLowerCase() === player.username.toLowerCase()) {
      var code = norm_(ref[i][0]);
      var ok = active[code] === true;
      if (ok) validCount++;
      codes.push({ code: code, valid: ok });
    }
  }
  codes.reverse(); // mới nhất lên đầu
  // xếp hạng theo số mã hợp lệ
  var arr = rankedList_();
  var rank = 0;
  for (var j = 0; j < arr.length; j++) {
    if (arr[j].username.toLowerCase() === player.username.toLowerCase()) { rank = j + 1; break; }
  }
  return { ok: true, username: player.username, validCount: validCount, pending: codes.length - validCount, rank: rank, codes: codes };
}

// ---------------- điểm vào web ----------------

function doGet(e) {
  var action = e && e.parameter && e.parameter.action;

  // App MeoMoney báo hoạt động thật: ?action=activity&code=XXXXXX
  if (action === 'activity') {
    recordActivity_(e.parameter.code);
    return json_({ ok: true });
  }

  // Không có action -> trả về trang web contest cho người chơi
  return HtmlService.createHtmlOutput(PAGE_HTML())
    .setTitle('Đua Top Giới Thiệu MeoMoney')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}
